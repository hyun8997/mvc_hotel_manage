CREATE TABLE HOTEL (
H_NO NUMBER(7) CONSTRAINT HOTEL_PK PRIMARY KEY,
AD_NO NUMBER(7),
H_NAME VARCHAR2(100),
LOC VARCHAR2(100),
H_CONV CLOB,
H_COMMENT CLOB);

CREATE SEQUENCE HOTEL_SEQ
INCREMENT BY 1 
START WITH 1 
MINVALUE 1 
MAXVALUE 1000 
NOCYCLE 
NOCACHE; 

CREATE TABLE ADMIN (
AD_NO NUMBER(7) CONSTRAINT ADMIN_PK PRIMARY KEY,
AD_ID VARCHAR2(100),
AD_PASSWORD VARCHAR2(100),
AD_NAME VARCHAR2(100),
AD_EMAIL VARCHAR2(100),
AD_TEL VARCHAR2(100),
AD_RANK NUMBER(1));

CREATE SEQUENCE ADMIN_SEQ
INCREMENT BY 1 
START WITH 1 
MINVALUE 1 
MAXVALUE 1000 
NOCYCLE 
NOCACHE;
